```typescript
import { Injectable, Logger, OnModuleDestroy } from '@nestjs/common';
import { ConfigService } from '@nestjs/config';

@Injectable()
export class Neo4jService implements OnModuleDestroy {
  private readonly logger = new Logger(Neo4jService.name);
  private driver: any;

  constructor(private configService: ConfigService) {
    this.initDriver();
  }

  private initDriver(): void {
    try {
      const neo4j = require('neo4j-driver');
      this.driver = neo4j.driver(
        this.configService.get('NEO4J_URI', 'bolt://localhost:7687'),
        neo4j.auth.basic(
          this.configService.get('NEO4J_USER', 'neo4j'),
          this.configService.get('NEO4J_PASSWORD', 'neo4j123')
        )
      );
      this.logger.log('Neo4j driver initialized');
    } catch (error) {
      this.logger.warn(`Neo4j driver init failed: ${error.message}`);
    }
  }

  async executeQuery(cypher: string, params: Record<string, any> = {}): Promise<any[]> {
    const session = this.driver.session();
    try {
      const result = await session.run(cypher, params);
      return result.records.map(record => {
        const obj: Record<string, any> = {};
        record.keys.forEach(key => { obj[key] = record.get(key); });
        return obj;
      });
    } finally {
      await session.close();
    }
  }

  async executeWrite(cypher: string, params: Record<string, any> = {}): Promise<void> {
    const session = this.driver.session();
    try {
      await session.writeTransaction(tx => tx.run(cypher, params));
    } finally {
      await session.close();
    }
  }

  async healthCheck(): Promise<boolean> {
    try {
      await this.executeQuery('RETURN 1 AS health');
      return true;
    } catch {
      return false;
    }
  }

  async onModuleDestroy(): Promise<void> {
    if (this.driver) {
      await this.driver.close();
    }
  }
}
```

Key design decisions:
- Uses ConfigService for env variables (not raw process.env)
- Parameterized queries only ($param syntax) — prevents Cypher injection
- Session-per-query pattern with proper cleanup in finally block
- Health check for graceful degradation

```typescript
async getRecommendationsByGraph(
  userId: string,
  occasion?: string,
  style?: string,
  topK: number = 20
): Promise<Array<{itemId: string, score: number, reasons: string[]}>> {
  let cypher: string;
  let params: Record<string, any> = { userId, topK };

  if (occasion && style) {
    cypher = `
      MATCH (user:User {id: $userId})-[:PREFERS]->(s:Style)
      MATCH (occ:Occasion {id: $occasion})-[:REQUIRES_STYLE]->(reqStyle:Style)
      MATCH (reqStyle)-[:COMPATIBLE_WITH*1..2]-(compStyle:Style)
      MATCH (item:ClothingItem)-[:HAS_STYLE]->(compStyle)
      MATCH (item)-[:SUITABLE_FOR]->(occ)
      WHERE NOT EXISTS((user)-[:VIEWED|DISLIKED]->(item))
      RETURN item.id AS itemId,
             count(DISTINCT compStyle) * 0.5 + count(DISTINCT s) * 0.3 AS score,
             collect(DISTINCT compStyle.id) AS matchedStyles
      ORDER BY score DESC
      LIMIT $topK
    `;
    params.occasion = occasion;
  } else if (style) {
    cypher = `
      MATCH (item:ClothingItem)-[:HAS_STYLE]->(s:Style {id: $style})
      MATCH (s)-[:COMPATIBLE_WITH*1..2]-(compStyle:Style)
      MATCH (compItem:ClothingItem)-[:HAS_STYLE]->(compStyle)
      RETURN compItem.id AS itemId,
             count(DISTINCT compStyle) AS score,
             collect(DISTINCT compStyle.id) AS matchedStyles
      ORDER BY score DESC
      LIMIT $topK
    `;
    params.style = style;
  } else {
    cypher = `
      MATCH (item:ClothingItem)-[:HAS_STYLE]->(s:Style)
      MATCH (item)-[:SUITABLE_FOR]->(o:Occasion)
      RETURN item.id AS itemId,
             count(DISTINCT s) + count(DISTINCT o) AS score,
             collect(DISTINCT s.id)[0..3] AS matchedStyles
      ORDER BY score DESC
      LIMIT $topK
    `;
  }

  const results = await this.neo4j.executeQuery(cypher, params);
  return results.map(r => ({
    itemId: r.itemId,
    score: Number(r.score),
    reasons: (r.matchedStyles || []).map((s: string) => `风格匹配: ${s}`),
  }));
}

private ciede2000(lab1: { L: number; a: number; b: number }, lab2: { L: number; a: number; b: number }): number {
  const L1 = lab1.L, a1 = lab1.a, b1 = lab1.b;
  const L2 = lab2.L, a2 = lab2.a, b2 = lab2.b;

  const C1 = Math.sqrt(a1 * a1 + b1 * b1);
  const C2 = Math.sqrt(a2 * a2 + b2 * b2);
  const Cab = (C1 + C2) / 2;

  const G = 0.5 * (1 - Math.sqrt(Math.pow(Cab, 7) / (Math.pow(Cab, 7) + Math.pow(25, 7))));
  const a1p = a1 * (1 + G);
  const a2p = a2 * (1 + G);

  const C1p = Math.sqrt(a1p * a1p + b1 * b1);
  const C2p = Math.sqrt(a2p * a2p + b2 * b2);

  const h1p = Math.atan2(b1, a1p) * 180 / Math.PI + (Math.atan2(b1, a1p) < 0 ? 360 : 0);
  const h2p = Math.atan2(b2, a2p) * 180 / Math.PI + (Math.atan2(b2, a2p) < 0 ? 360 : 0);

  const dLp = L2 - L1;
  const dCp = C2p - C1p;

  let dhp: number;
  if (C1p * C2p === 0) { dhp = 0; }
  else if (Math.abs(h2p - h1p) <= 180) { dhp = h2p - h1p; }
  else if (h2p - h1p > 180) { dhp = h2p - h1p - 360; }
  else { dhp = h2p - h1p + 360; }

  const dHp = 2 * Math.sqrt(C1p * C2p) * Math.sin(dhp * Math.PI / 360);

  const Lbp = (L1 + L2) / 2;
  const Cbp = (C1p + C2p) / 2;

  let hbp: number;
  if (C1p * C2p === 0) { hbp = h1p + h2p; }
  else if (Math.abs(h1p - h2p) <= 180) { hbp = (h1p + h2p) / 2; }
  else if (h1p + h2p < 360) { hbp = (h1p + h2p + 360) / 2; }
  else { hbp = (h1p + h2p - 360) / 2; }

  const T = 1
    - 0.17 * Math.cos((hbp - 30) * Math.PI / 180)
    + 0.24 * Math.cos(2 * hbp * Math.PI / 180)
    + 0.32 * Math.cos((3 * hbp + 6) * Math.PI / 180)
    - 0.20 * Math.cos((4 * hbp - 63) * Math.PI / 180);

  const SL = 1 + 0.015 * Math.pow(Lbp - 50, 2) / Math.sqrt(20 + Math.pow(Lbp - 50, 2));
  const SC = 1 + 0.045 * Cbp;
  const SH = 1 + 0.015 * Cbp * T;

  const RT = -Math.sin(2 * (30 * Math.PI / 180))
    * 2 * Math.sqrt(Math.pow(Cbp, 7) / (Math.pow(Cbp, 7) + Math.pow(25, 7)));

  const dE = Math.sqrt(
    Math.pow(dLp / SL, 2)
    + Math.pow(dCp / SC, 2)
    + Math.pow(dHp / SH, 2)
    + RT * (dCp / SC) * (dHp / SH)
  );

  return dE;
}
```

calculatePerceptualHarmonyScore(
  colors: string[],
  userColorSeason?: string
): { score: number; details: { rule: string; contribution: number }[] } {
  if (colors.length < 2) return { score: 50, details: [] };

  const labColors = colors.map(c => this.rgbToLab(this.hexToRgb(c)));
  const details: { rule: string; contribution: number }[] = [];
  let totalScore = 0;
  let pairCount = 0;

  // Pairwise CIEDE2000 distance scoring
  for (let i = 0; i < labColors.length; i++) {
    for (let j = i + 1; j < labColors.length; j++) {
      const de = this.ciede2000(labColors[i], labColors[j]);
      // CIEDE2000: 0 = identical, 1 = just noticeable, 5 = significant, 10+ = very different
      let pairScore: number;
      if (de < 5) pairScore = 90; // Very similar — monochromatic harmony
      else if (de < 15) pairScore = 80; // Analogous
      else if (de < 30) pairScore = 70; // Moderate contrast
      else if (de < 50) pairScore = 60; // Complementary range
      else pairScore = 40; // Too contrasting
      totalScore += pairScore;
      pairCount++;
      details.push({ rule: `CIEDE2000 pair (${i+1},${j+1}): ΔE=${de.toFixed(1)}`, contribution: pairScore });
    }
  }

  let score = pairCount > 0 ? totalScore / pairCount : 50;

  // Color season bonus (D-16: personal color season fusion)
  if (userColorSeason) {
    const seasonBonus = this.getColorSeasonCompatibility(colors, userColorSeason);
    score = score * 0.7 + seasonBonus * 0.3;
    details.push({ rule: `Color season (${userColorSeason}) compatibility`, contribution: seasonBonus });
  }

  return { score: Math.round(Math.min(100, Math.max(0, score))), details };
}
```

  const palette = seasonPalettes[season] || seasonPalettes.spring_warm;

class SASRec(nn.Module):
    def __init__(self, item_num: int, hidden_units: int = 64, max_len: int = 50,
                 num_heads: int = 2, num_blocks: int = 2, dropout: float = 0.2):
        super().__init__()
        self.item_num = item_num
        self.hidden_units = hidden_units
        self.max_len = max_len

        self.item_emb = nn.Embedding(item_num + 1, hidden_units, padding_idx=0)
        self.pos_emb = nn.Embedding(max_len, hidden_units)
        self.emb_dropout = nn.Dropout(dropout)

        self.transformer_blocks = nn.ModuleList([
            nn.TransformerEncoderLayer(
                d_model=hidden_units,
                nhead=num_heads,
                dim_feedforward=hidden_units * 4,
                dropout=dropout,
                activation='gelu',
                batch_first=True,
            )
            for _ in range(num_blocks)
        ])

        self.layer_norm = nn.LayerNorm(hidden_units)

    def forward(self, seq: torch.Tensor) -> torch.Tensor:
        batch_size = seq.size(0)
        positions = torch.arange(seq.size(1), device=seq.device).unsqueeze(0).expand(batch_size, -1)

        x = self.item_emb(seq) + self.pos_emb(positions)
        x = self.emb_dropout(x)

        mask = (seq == 0).unsqueeze(-1).expand_as(x)
        for block in self.transformer_blocks:
            x = block(x, src_key_padding_mask=seq == 0)

        x = self.layer_norm(x)
        return x[:, -1, :]

    def predict(self, seq: torch.Tensor) -> torch.Tensor:
        user_emb = self.forward(seq)
        item_embs = self.item_emb.weight[1:]
        scores = torch.matmul(user_emb, item_embs.transpose(0, 1))
        return scores
```

    start = time.time()
    try:
        with torch.no_grad():
            seq = torch.tensor([req.item_sequence[-MAX_LEN:]]).long()
            scores = model.predict(seq)
            top_k = min(req.top_k, scores.size(1))

            recommendations=item_ids,

```typescript
import { Injectable, Logger } from '@nestjs/common';
import { ConfigService } from '@nestjs/config';

interface SASRecRecommendRequest {
  user_id: string;
  item_sequence: number[];
  top_k: number;
  include_scores: boolean;
}

interface SASRecRecommendResponse {
  success: boolean;
  recommendations: number[];
  scores: number[] | null;
  processing_time: number;
}

@Injectable()
export class SasrecClientService {
  private readonly logger = new Logger(SasrecClientService.name);
  private readonly serviceUrl: string;
  private readonly timeoutMs: number;

  constructor(private configService: ConfigService) {
    this.serviceUrl = this.configService.get('SASREC_SERVICE_URL', 'http://localhost:8001');
    this.timeoutMs = this.configService.get<number>('SASREC_TIMEOUT_MS', 5000);
  }

  async getRecommendations(
    userId: string,
    itemSequence: number[],
    topK: number = 10
  ): Promise<Array<{ itemId: string; score: number }>> {
    try {
      const controller = new AbortController();
      const timeout = setTimeout(() => controller.abort(), this.timeoutMs);

      const response = await fetch(`${this.serviceUrl}/api/recommend`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          user_id: userId,
          item_sequence: itemSequence,
          top_k: topK,
          include_scores: true,
        } as SASRecRecommendRequest),
        signal: controller.signal,
      });

      clearTimeout(timeout);

      if (!response.ok) {
        throw new Error(`SASRec service returned ${response.status}`);
      }

      const data: SASRecRecommendResponse = await response.json();
      if (!data.success) {
        throw new Error('SASRec service returned unsuccessful response');
      }

      return data.recommendations.map((itemId, idx) => ({
        itemId: String(itemId),
        score: data.scores ? data.scores[idx] : 0,
      }));
    } catch (error) {
      this.logger.warn(`SASRec service unavailable: ${error.message}`);
      return [];
    }
  }

  async healthCheck(): Promise<boolean> {
    try {
      const response = await fetch(`${this.serviceUrl}/health`, {
        signal: AbortSignal.timeout(3000),
      });
      return response.ok;
    } catch {
      return false;
    }
  }
}
```

- Uses native `fetch()` (Node 18+) instead of axios
- Configurable timeout via SASREC_TIMEOUT_MS env var
- Graceful degradation — returns empty array on failure instead of throwing
- Health check for monitoring
