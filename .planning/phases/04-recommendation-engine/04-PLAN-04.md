```typescript
private calculateTimeDecay(createdAt: Date, halfLifeHours: number = 168): number {
  const ageHours = (Date.now() - createdAt.getTime()) / (1000 * 60 * 60);
  return Math.pow(0.5, ageHours / halfLifeHours);
}
```
- halfLifeHours = 168 (7 days) — behaviors older than 7 days have half the weight
- This implements exponential decay: weight = 0.5^(age/halfLife)

```typescript
async getHybridRecommendations(userId: string, topK: number = 10): Promise<Array<{itemId: string; score: number; reason: string}>> {
  const profile = await this.prisma.userProfile.findUnique({ where: { userId } });
  const hasProfile = !!profile;
  const hasColorSeason = !!profile?.colorSeason;
  const hasBodyType = !!profile?.bodyType;

  const [demographic, popular] = await Promise.all([
    this.getDemographicRecommendations(userId, topK),
    this.getPopularRecommendations(topK),
  ]);

  const weights = {
    demographic: hasProfile ? 0.5 : 0.2,
    popular: hasProfile ? 0.3 : 0.6,
    quiz: hasProfile ? 0.2 : 0.2,
  };

  const merged = new Map<string, {score: number, reason: string}>();
  for (const item of demographic) {
    merged.set(item.itemId, { score: item.score * weights.demographic, reason: item.reason });
  }
  for (const item of popular) {
    const existing = merged.get(item.itemId);
    if (existing) {
      existing.score += item.score * weights.popular;
    } else {
      merged.set(item.itemId, { score: item.score * weights.popular, reason: item.reason });
    }
  }

  return Array.from(merged.entries())
    .map(([itemId, data]) => ({ itemId, ...data }))
    .sort((a, b) => b.score - a.score)
    .slice(0, topK);
}
```

**1. Implement `generateReason()` method** using GLM API (D-12: GLM only for reason text):
```typescript
async generateReason(params: {
  userId: string;
  itemId: string;
  scores: Record<string, number>;
  userProfile?: any;
  itemDetails?: any;
  colorHarmony?: { score: number; details: any[] };
}): Promise<string> {
  const { scores, userProfile, itemDetails, colorHarmony } = params;

  // Build context for GLM prompt
  const reasons: string[] = [];

  if (scores.ruleScore > 70) {
    if (userProfile?.bodyType && itemDetails?.attributes?.bodyTypeFit?.includes(userProfile.bodyType)) {
      reasons.push(`适合你的${userProfile.bodyType}体型`);
    }
  }

  if (scores.colorScore > 60 && userProfile?.colorSeason) {
    reasons.push(`适合你的${this.translateColorSeason(userProfile.colorSeason)}色彩`);
  }

  if (colorHarmony && colorHarmony.score > 70) {
    reasons.push(`色彩搭配和谐度${colorHarmony.score}%`);
  }

  if (scores.cfScore > 0.5) {
    reasons.push('与你品味相似的用户也喜欢');
  }

  if (scores.kgScore > 0.5) {
    reasons.push('风格搭配推荐');
  }

  // If we have enough rule-based reasons, skip GLM call (cost optimization)
  if (reasons.length >= 2) {
    return reasons.slice(0, 2).join('，');
  }

  // Fallback to GLM for personalized reason generation
  try {
    const prompt = `为以下穿搭推荐生成1句话的推荐理由（15字以内）：
用户画像：${JSON.stringify({ gender: userProfile?.gender, colorSeason: userProfile?.colorSeason, bodyType: userProfile?.bodyType })}
商品：${JSON.stringify({ name: itemDetails?.name, category: itemDetails?.category, style: itemDetails?.attributes?.style })}
评分：${JSON.stringify(scores)}
色彩和谐度：${colorHarmony?.score || 'N/A'}%

推荐理由：`;

    const glmResponse = await this.callGLM(prompt);
    return glmResponse || reasons.join('，') || '为你精选推荐';
  } catch {
    return reasons.join('，') || '为你精选推荐';
  }
}
```

**2. Add `callGLM()` private method**:
```typescript
private async callGLM(prompt: string): Promise<string | null> {
  const apiKey = this.configService.get('GLM_API_KEY');
  if (!apiKey) return null;

  try {
    const response = await fetch('https://open.bigmodel.cn/api/paas/v4/chat/completions', {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${apiKey}`,
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({
        model: 'glm-4-flash',
        messages: [{ role: 'user', content: prompt }],
        max_tokens: 50,
        temperature: 0.7,
      }),
    });

    const data = await response.json();
    return data.choices?.[0]?.message?.content?.trim() || null;
  } catch {
    return null;
  }
}
```

```typescript
import { Injectable, Logger } from '@nestjs/common';
import { PrismaService } from '../../../common/services/prisma.service';
import { UnifiedRecommendationEngine } from './unified-recommendation.engine';
import { RecommendationExplainerService } from './recommendation-explainer.service';
import { ColorMatchingService } from './color-matching.service';
import { ColdStartService } from './cold-start.service';
import { CacheService } from '../../../common/services/cache.service';

interface FeedItem {
  id: string;
  mainImage: string;
  brand: { id: string; name: string } | null;
  price: number;
  originalPrice?: number;
  styleTags: string[];
  colorHarmony: { score: number; colors: string[] };
  matchReason: string;
  category: string;
}

interface FeedResult {
  items: FeedItem[];
  total: number;
  hasMore: boolean;
}

@Injectable()
export class RecommendationFeedService {
  private readonly logger = new Logger(RecommendationFeedService.name);

  constructor(
    private prisma: PrismaService,
    private engine: UnifiedRecommendationEngine,
    private explainer: RecommendationExplainerService,
    private colorMatching: ColorMatchingService,
    private coldStart: ColdStartService,
    private cacheService: CacheService,
  ) {}

  async getFeed(
    userId: string,
    category: 'daily' | 'occasion' | 'trending' | 'explore',
    subCategory?: string,
    page: number = 1,
    pageSize: number = 10,
  ): Promise<FeedResult> {
    // Check cache first
    const cacheKey = `rec:${userId}:${category}:${subCategory || 'all'}:${page}`;
    const cached = await this.cacheService.get<FeedResult>(cacheKey);
    if (cached) return cached;

    // Get recommendations from engine
    let recommendations: Array<{itemId: string; score: number; reasons?: string[]}>;

    const userBehaviorCount = await this.prisma.userBehavior.count({
      where: { userId },
    });

    if (userBehaviorCount < 5) {
      // Cold start path
      recommendations = await this.coldStart.getHybridRecommendations(userId, pageSize * 2);
    } else {
      // Normal path — use unified engine
      recommendations = await this.engine.recommend(userId, {
        category,
        subCategory,
        topK: pageSize * 2,
      });
    }

    // Enrich items with details
    const items = await this.enrichItems(userId, recommendations, page, pageSize);

    const result: FeedResult = {
      items,
      total: recommendations.length,
      hasMore: (page * pageSize) < recommendations.length,
    };

    // Cache result
    await this.cacheService.set(cacheKey, result, 3600); // 1 hour TTL

    return result;
  }

  private async enrichItems(
    userId: string,
    recommendations: Array<{itemId: string; score: number; reasons?: string[]}>,
    page: number,
    pageSize: number,
  ): Promise<FeedItem[]> {
    const pagedRecs = recommendations.slice((page - 1) * pageSize, page * pageSize);
    if (pagedRecs.length === 0) return [];

    const itemIds = pagedRecs.map(r => r.itemId);
    const clothingItems = await this.prisma.clothingItem.findMany({
      where: { id: { in: itemIds } },
      include: { brand: true },
    });

    const itemMap = new Map(clothingItems.map(i => [i.id, i]));
    const userProfile = await this.prisma.userProfile.findUnique({ where: { userId } });

    const feedItems: FeedItem[] = [];
    for (const rec of pagedRecs) {
      const item = itemMap.get(rec.itemId);
      if (!item) continue;

      // Calculate color harmony score
      const itemColors = item.attributes?.colors || [item.attributes?.primaryColor].filter(Boolean);
      const colorHarmonyResult = this.colorMatching.calculatePerceptualHarmonyScore(
        itemColors,
        userProfile?.colorSeason || undefined,
      );

      // Get match reason
      const matchReason = rec.reasons?.[0] || await this.explainer.generateReason({
        userId,
        itemId: item.id,
        scores: { ruleScore: rec.score * 100 },
        userProfile: userProfile || undefined,
        colorHarmony: colorHarmonyResult,
      });

      feedItems.push({
        id: item.id,
        mainImage: item.mainImage,
        brand: item.brand ? { id: item.brand.id, name: item.brand.name } : null,
        price: Number(item.price),
        originalPrice: item.originalPrice ? Number(item.originalPrice) : undefined,
        styleTags: (item.attributes?.style || []).slice(0, 2),
        colorHarmony: {
          score: colorHarmonyResult.score,
          colors: itemColors.slice(0, 3),
        },
        matchReason,
        category: item.category,
      });
    }

    return feedItems;
  }
}
```
