Alternative approach (HOC wrapper):
```typescript
// Create a guarded screen factory
function withGuards(Component: React.ComponentType, guards: { auth?: boolean; vip?: boolean }) {
  return () => {
    let wrapped = <Component />;
    if (guards.vip) wrapped = <VipGuard>{wrapped}</VipGuard>;
    if (guards.auth) wrapped = <AuthGuard>{wrapped}</AuthGuard>;
    return wrapped;
  };
}

// Use in navigator
<Stack.Screen name="CustomDesign" component={withGuards(CustomDesignScreen, { auth: true, vip: true })} />
```
