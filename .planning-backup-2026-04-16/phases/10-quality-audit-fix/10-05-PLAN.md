```typescript
/**
 * Interaction-semantic spring configurations.
 * Maps user interaction types to appropriate animation feel.
 * 
 * Usage:
 * const spring = interactionSprings.confirm; // Quick, satisfying
 * const spring = interactionSprings.error;   // Stiff, attention-grabbing
 */
export const interactionSprings = {
  // Light operations - quick, snappy feedback
  tap: springConfigs.snappy,          // Button taps, selections
  toggle: springConfigs.snappy,       // Switches, checkboxes
  
  // Medium operations - gentle, comfortable transitions
  slide: springConfigs.gentle,        // Card slides, drawer reveals
  expand: springConfigs.gentle,       // Accordion, collapse
  navigate: springConfigs.gentle,     // Page transitions
  
  // Heavy operations - bouncy, celebratory
  confirm: springConfigs.bouncy,      // Purchase complete, save success
  celebrate: springConfigs.bouncy,    // Achievement unlocked, milestone
  like: springConfigs.bouncy,         // Heart, favorite
  
  // Error/Alert - stiff, attention-demanding
  error: springConfigs.stiff,         // Validation error, shake
  alert: springConfigs.stiff,         // Important notification
  warning: springConfigs.stiff,       // Destructive action confirm
};

/**
 * Get semantic spring config with reduced motion support.
 * @param interaction - Type of user interaction
 * @param reducedMotion - Whether user prefers reduced motion
 */
export function getInteractionSpring(
  interaction: keyof typeof interactionSprings,
  reducedMotion: boolean
) {
  if (reducedMotion) {
    return reducedMotionSpring;
  }
  return interactionSprings[interaction];
}

/**
 * Semantic duration mappings.
 */
export const interactionDurations = {
  feedback: durations.fast,       // Micro-feedback (ripple, press)
  transition: durations.normal,   // Page/modal transitions
  reveal: durations.slow,         // Content reveal, loading complete
  attention: durations.verySlow,  // Attention-demanding animations
};
```

This creates a clear mapping:
- snappy -> light operations (tap, toggle)
- gentle -> medium operations (slide, navigate)  
- bouncy -> heavy/success operations (confirm, celebrate)
- stiff -> error/alert states

import React, { useEffect, useState } from 'react';
import { View, Text, StyleSheet } from 'react-native';
import Animated, {
  useSharedValue,
  useAnimatedStyle,
  withTiming,
  withSequence,
  withDelay,
  interpolateColor,
  Easing,
} from 'react-native-reanimated';
import { useReducedMotion } from '../../hooks/useReducedMotion';
import { DesignTokens } from '../../theme/tokens/design-tokens';

const TERRACOTTA = DesignTokens.colors.brand.primary; // #C67B5C

interface AIThinkingAnimationProps {
  phase?: 'analyzing' | 'selecting' | 'completing';
}

export const AIThinkingAnimation: React.FC<AIThinkingAnimationProps> = ({ phase = 'analyzing' }) => {
  const reducedMotion = useReducedMotion();
  const [currentPhase, setCurrentPhase] = useState(0);
  
  const lineProgress = useSharedValue(0);
  const silhouetteOpacity = useSharedValue(0);
  const outfitOpacity = useSharedValue(0);
  
  useEffect(() => {
    if (reducedMotion) {
      // In reduced motion: just show text phases
      const timer1 = setTimeout(() => setCurrentPhase(1), 2000);
      const timer2 = setTimeout(() => setCurrentPhase(2), 4000);
      return () => { clearTimeout(timer1); clearTimeout(timer2); };
    }
    
    // Phase 1: Flowing lines
    lineProgress.value = withTiming(1, { duration: 2000, easing: Easing.inOut(Easing.ease) });
    
    // Phase 2: Silhouette appears
    silhouetteOpacity.value = withDelay(2000, withTiming(1, { duration: 1000 }));
    
    // Phase 3: Full outfit
    outfitOpacity.value = withDelay(4000, withTiming(1, { duration: 1500 }));
    
    // Update text phases
    const timer1 = setTimeout(() => setCurrentPhase(1), 2000);
    const timer2 = setTimeout(() => setCurrentPhase(2), 4000);
    
    return () => { clearTimeout(timer1); clearTimeout(timer2); };
  }, [reducedMotion]);
  
  const phaseTexts = ['正在分析您的风格...', '正在挑选单品...', '正在搭配完整造型...'];
  
  const lineStyle = useAnimatedStyle(() => ({
    width: `${lineProgress.value * 100}%`,
    height: 3,
    backgroundColor: TERRACOTTA,
    borderRadius: 1.5,
  }));
  
  const silhouetteStyle = useAnimatedStyle(() => ({
    opacity: silhouetteOpacity.value,
  }));
  
  const outfitStyle = useAnimatedStyle(() => ({
    opacity: outfitOpacity.value,
  }));
  
  if (reducedMotion) {
    return (
      <View style={styles.container}>
        <View style={styles.progressDots}>
          {[0, 1, 2].map((i) => (
            <View key={i} style={[styles.dot, currentPhase >= i && styles.dotActive]} />
          ))}
        </View>
        <Text style={styles.phaseText}>{phaseTexts[currentPhase]}</Text>
      </View>
    );
  }
  
  return (
    <View style={styles.container}>
      {/* Phase 1: Flowing gradient line */}
      <View style={styles.lineContainer}>
        <Animated.View style={lineStyle} />
      </View>
      
      {/* Phase 2: Silhouette */}
      <Animated.View style={[styles.silhouette, silhouetteStyle]}>
        {/* Simple clothing icon or SVG silhouette */}
        <View style={styles.silhouetteIcon} />
      </Animated.View>
      
      {/* Phase 3: Outfit preview placeholder */}
      <Animated.View style={[styles.outfitPreview, outfitStyle]}>
        <View style={styles.outfitPlaceholder} />
      </Animated.View>
      
      <Text style={styles.phaseText}>{phaseTexts[currentPhase]}</Text>
    </View>
  );
};

const styles = StyleSheet.create({

  silhouette: { width: 60, height: 80, marginBottom: 12 },
  silhouetteIcon: { width: '100%', height: '100%', backgroundColor: '#F0E6E0', borderRadius: 8 },

  outfitPlaceholder: { width: '100%', height: '100%', backgroundColor: '#FAF0EB', borderRadius: 12 },

  dot: { width: 8, height: 8, borderRadius: 4, backgroundColor: '#E0E0E0' },
  dotActive: { backgroundColor: '#C67B5C' },
});

export default AIThinkingAnimation;
