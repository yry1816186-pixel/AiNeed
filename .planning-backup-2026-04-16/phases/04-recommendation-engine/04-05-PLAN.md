```typescript
type RecommendationTab = 'daily' | 'occasion' | 'trending' | 'explore';
type OccasionType = 'commute' | 'date' | 'sport' | 'interview' | 'casual' | 'travel';

interface RecommendationState {
  activeTab: RecommendationTab;
  activeOccasion: OccasionType;
  pendingFeedback: Map<string, 'like' | 'dislike'>;
  setActiveTab: (tab: RecommendationTab) => void;
  setActiveOccasion: (occasion: OccasionType) => void;
  addFeedback: (itemId: string, action: 'like' | 'dislike') => void;
  clearFeedback: () => void;
}

export const useRecommendationStore = create<RecommendationState>((set) => ({
  activeTab: 'daily',
  activeOccasion: 'commute',
  pendingFeedback: new Map(),
  setActiveTab: (tab) => set({ activeTab: tab }),
  setActiveOccasion: (occasion) => set({ activeOccasion: occasion }),
  addFeedback: (itemId, action) =>
    set((state) => {
      const updated = new Map(state.pendingFeedback);
      updated.set(itemId, action);
      return { pendingFeedback: updated };
    }),
  clearFeedback: () => set({ pendingFeedback: new Map() }),
}));
```

```typescript
  price: number;
  originalPrice?: number;
  styleTags: string[];
  colorHarmony: { score: number; colors: string[] };

  items: FeedItem[];
  total: number;
  hasMore: boolean;
}

export interface FeedParams {
  category: 'daily' | 'occasion' | 'trending' | 'explore';
  subCategory?: string;
  page?: number;
  pageSize?: number;
}

export const recommendationService = {
  getFeed: async (params: FeedParams): Promise<FeedResponse> => {
    const { category, subCategory, page = 1, pageSize = 10 } = params;
    const queryParams: Record<string, string> = {
      category,
      page: String(page),
      pageSize: String(pageSize),
    };
    if (subCategory) queryParams.subCategory = subCategory;

    const response = await apiClient.get<FeedResponse>('/api/v1/recommendations/feed', {
      params: queryParams,
    });
    return response.data;
  },

  trackBehavior: async (itemId: string, action: string): Promise<void> => {
    await apiClient.post('/api/v1/recommendations/feedback', { itemId, action });
  },
};
```

```typescript
import Animated, { useSharedValue, useAnimatedProps, withTiming, withDelay } from 'react-native-reanimated';
import Svg, { Circle } from 'react-native-svg';

const AnimatedCircle = Animated.createAnimatedComponent(Circle);

interface ColorScoreRingProps {
  score: number;
  colors: string[];
  size?: number;
  strokeWidth?: number;
}

export const ColorScoreRing: React.FC<ColorScoreRingProps> = ({
  score,
  colors,
  size = 28,
  strokeWidth = 3,
}) => {
  const progress = useSharedValue(0);
  const radius = (size - strokeWidth) / 2;

    strokeDashoffset: circumference * (1 - progress.value),

        <View key={index} style={[styles.colorBlock, { backgroundColor: color }]} />
```

```typescript
import React, { useCallback } from 'react';
import { View, Text, StyleSheet } from 'react-native';
import {
  Gesture,
  GestureDetector,
  SimultaneousGestures,
} from 'react-native-gesture-handler';
import Animated, {
  useSharedValue,
  useAnimatedStyle,
  withSpring,
  withTiming,
  runOnJS,
  interpolate,
} from 'react-native-reanimated';

interface SwipeableCardWrapperProps {
  children: React.ReactNode;
  onLike?: () => void;
  onDislike?: () => void;
  swipeThreshold?: number;
}

export const SwipeableCardWrapper: React.FC<SwipeableCardWrapperProps> = ({
  children,
  onLike,
  onDislike,
  swipeThreshold = 60,
}) => {
  const translateX = useSharedValue(0);
  const cardOpacity = useSharedValue(1);

  const handleLike = useCallback(() => onLike?.(), [onLike]);
  const handleDislike = useCallback(() => onDislike?.(), [onDislike]);

  const panGesture = Gesture.Pan()
    .activeOffsetX([-10, 10])
    .failOffsetY([-10, 10])
    .onUpdate((event) => {
      translateX.value = event.translationX;
    })
    .onEnd(() => {
      if (Math.abs(translateX.value) > swipeThreshold) {
        const callback = translateX.value > 0 ? handleLike : handleDislike;
        translateX.value = withTiming(translateX.value > 0 ? 300 : -300, { duration: 300 });
        cardOpacity.value = withTiming(0, { duration: 300 }, (finished) => {
          if (finished) {
            runOnJS(callback)();
            translateX.value = 0;
            cardOpacity.value = 1;
          }
        });
      } else {
        translateX.value = withSpring(0, { damping: 15, stiffness: 200 });
      }
    });

  const animatedStyle = useAnimatedStyle(() => ({
    transform: [
      { translateX: translateX.value },
      { rotate: `${interpolate(translateX.value, [-120, 0, 120], [-0.08, 0, 0.08])}rad` },
    ],
    opacity: cardOpacity.value,
  }));

  const likeOverlayStyle = useAnimatedStyle(() => ({
    opacity: interpolate(translateX.value, [0, 20, 60], [0, 0.5, 1]),
    backgroundColor: 'rgba(91,138,114,0.12)',
  }));

  const dislikeOverlayStyle = useAnimatedStyle(() => ({
    opacity: interpolate(translateX.value, [-60, -20, 0], [1, 0.5, 0]),
    backgroundColor: 'rgba(196,69,54,0.12)',
  }));

  return (
    <GestureDetector gesture={panGesture}>
      <Animated.View style={[styles.container, animatedStyle]}>
        {children}
        <Animated.View style={[styles.overlay, likeOverlayStyle]} pointerEvents="none">
          <Text style={styles.likeText}>喜欢</Text>
        </Animated.View>
        <Animated.View style={[styles.overlay, dislikeOverlayStyle]} pointerEvents="none">
          <Text style={styles.dislikeText}>不喜欢</Text>
        </Animated.View>
      </Animated.View>
    </GestureDetector>
  );
};

const styles = StyleSheet.create({
  container: {
    position: 'relative',
  },
  overlay: {
    position: 'absolute',
    top: 0,
    left: 0,
    right: 0,
    bottom: 0,
    justifyContent: 'center',
    alignItems: 'center',
    borderRadius: 12,
  },
  likeText: {
    fontSize: 18,
    fontWeight: '700',
    color: '#5B8A72',
  },
  dislikeText: {
    fontSize: 18,
    fontWeight: '700',
    color: '#C44536',
  },
});
```

Key design decisions per UI-SPEC:
- `activeOffsetX` and `failOffsetY` ensure horizontal swipe doesn't conflict with vertical scroll
- 60px threshold for trigger (D-08)
- Spring snap-back with damping:15, stiffness:200
- Slight rotation: interpolate(translateX, [-120, 0, 120], [-0.08, 0, 0.08])
- Like overlay: green (#5B8A72), Dislike overlay: red (#C44536)
- After swipe off-screen, callback fires, then card resets for reuse

```typescript
import React, { useCallback } from 'react';
import { View, StyleSheet, RefreshControl } from 'react-native';
import { MasonryFlashList } from '@shopify/flash-list';
import { useInfiniteQuery, useQueryClient } from '@tanstack/react-query';
import { useRecommendationStore } from '../../stores/recommendationStore';
import { recommendationService, FeedItem } from '../../services/recommendationService';
import { RecommendationTabBar } from './RecommendationTabBar';
import { OccasionSubTabs } from './OccasionSubTabs';
import { RecommendationCard } from './RecommendationCard';
import { SwipeableCardWrapper } from './SwipeableCardWrapper';
import { RecommendationEmptyState } from './RecommendationEmptyState';
import { RecommendationSkeleton } from './RecommendationSkeleton';

export const RecommendationFeed: React.FC = () => {
  const { activeTab, activeOccasion, addFeedback } = useRecommendationStore();
  const queryClient = useQueryClient();

  const { data, fetchNextPage, hasNextPage, isFetchingNextPage, isLoading, refetch } =
    useInfiniteQuery({
      queryKey: ['recommendations', activeTab, activeOccasion],
      queryFn: ({ pageParam = 1 }) =>
        recommendationService.getFeed({
          category: activeTab,
          subCategory: activeTab === 'occasion' ? activeOccasion : undefined,
          page: pageParam,
          pageSize: 10,
        }),
      getNextPageParam: (lastPage, allPages) =>
        lastPage.hasMore ? allPages.length + 1 : undefined,
      initialPageParam: 1,
    });

  const items: FeedItem[] = data?.pages.flatMap((page) => page.items) ?? [];

  const handleLike = useCallback((itemId: string) => {
    addFeedback(itemId, 'like');
    recommendationService.trackBehavior(itemId, 'like');
  }, [addFeedback]);

  const handleDislike = useCallback((itemId: string) => {
    addFeedback(itemId, 'dislike');
    recommendationService.trackBehavior(itemId, 'dislike');
  }, [addFeedback]);

  const renderItem = useCallback(({ item }: { item: FeedItem }) => (
    <SwipeableCardWrapper
      onLike={() => handleLike(item.id)}
      onDislike={() => handleDislike(item.id)}
    >
      <RecommendationCard
        mainImage={item.mainImage}
        brand={item.brand}
        price={item.price}
        originalPrice={item.originalPrice}
        styleTags={item.styleTags}
        colorHarmony={item.colorHarmony}
        matchReason={item.matchReason}
      />
    </SwipeableCardWrapper>
  ), [handleLike, handleDislike]);

  if (isLoading) return <RecommendationSkeleton />;

  if (items.length === 0) return (
    <View style={styles.container}>
      <RecommendationTabBar />
      <OccasionSubTabs />
      <RecommendationEmptyState />
    </View>
  );

  return (
    <View style={styles.container}>
      <RecommendationTabBar />
      <OccasionSubTabs />
      <MasonryFlashList
        data={items}
        renderItem={renderItem}
        estimatedItemSize={280}
        numColumns={2}
        onEndReached={() => {
          if (hasNextPage && !isFetchingNextPage) fetchNextPage();
        }}
        onEndReachedThreshold={0.5}
        refreshControl={
          <RefreshControl
            refreshing={isLoading}
            onRefresh={() => refetch()}
            tintColor="#C67B5C"
          />
        }
        contentContainerStyle={styles.listContent}
      />
    </View>
  );
};

const styles = StyleSheet.create({
  container: {
    flex: 1,
    backgroundColor: '#FAFAF8',
  },
  listContent: {
    paddingHorizontal: 20,
    paddingTop: 8,
  },
});
```
